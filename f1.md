# Set 3 — Part A (Count occurrences of character "A" typed in textfield)

`MainActivity.java`

```java
package com.example.set3a;

import android.os.Bundle;
import android.text.Editable;
import android.text.TextWatcher;
import android.widget.EditText;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    int count = 0;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        EditText et = findViewById(R.id.etText);
        TextView tvCount = findViewById(R.id.tvCount);
        et.addTextChangedListener(new TextWatcher() {
            String prev = "";
            @Override
            public void beforeTextChanged(CharSequence s,int start,int count,int after){ prev = s.toString(); }
            @Override
            public void onTextChanged(CharSequence s,int start,int before,int count){}
            @Override
            public void afterTextChanged(Editable s){
                String now = s.toString();
                if (now.length()>prev.length()){
                    char added = now.charAt(now.length()-1);
                    if (added=='A' || added=='a'){ MainActivity.this.count++; tvCount.setText("A count: "+MainActivity.this.count); }
                } else if (now.length()<prev.length()){
                    int removed = prev.length() - now.length();
                    for (int i=0;i<removed;i++){
                        int pos = prev.length()-1-i;
                        char c = prev.charAt(pos);
                        if (c=='A' || c=='a'){ MainActivity.this.count = Math.max(0, MainActivity.this.count-1); }
                    }
                    tvCount.setText("A count: "+MainActivity.this.count);
                }
                prev = now;
            }
        });
    }
}
```

```java
    int enterCount = 0;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        EditText et = findViewById(R.id.etInput);
        TextView tv = findViewById(R.id.tvCount);
        et.setOnKeyListener((v,keyCode,event) -> {
            if (keyCode==KeyEvent.KEYCODE_ENTER && event.getAction()==KeyEvent.ACTION_DOWN){ enterCount++; tv.setText("Enter pressed: "+enterCount); return true; }
            return false;
        });
        et.setOnEditorActionListener((v,actionId,event) -> {
            if (actionId==EditorInfo.IME_ACTION_DONE){ enterCount++; tv.setText("Enter pressed: "+enterCount); return true; }
            return false;
        });
    }
}
```

```java
        });
    }
}
```

```java
package com.example.set5b;

import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.database.sqlite.SQLiteOpenHelper;
import android.os.Bundle;
import android.text.InputType;
import android.widget.ArrayAdapter;
import android.widget.Button;
import android.widget.EditText;
import android.widget.ListView;
import android.widget.Spinner;
import android.widget.Toast;
import androidx.appcompat.app.AppCompatActivity;
import java.util.ArrayList;

public class MainActivity extends AppCompatActivity {
    DBHelper db;
    ArrayAdapter<String> adapter;
    ArrayList<String> list = new ArrayList<>();
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        db = new DBHelper();
        EditText etName = findViewById(R.id.etName);
        Spinner spinnerGender = findViewById(R.id.spinnerGender);
        EditText etAddress = findViewById(R.id.etAddress);
        EditText etNationality = findViewById(R.id.etNationality);
        EditText etMobile = findViewById(R.id.etMobile);
        Button btnSave = findViewById(R.id.btnSave);
        ListView lv = findViewById(R.id.listView);
        etMobile.setInputType(InputType.TYPE_CLASS_PHONE);
        ArrayAdapter<String> gAdapter = new ArrayAdapter<>(this, android.R.layout.simple_spinner_item, new String[]{"Male","Female","Other"});
        gAdapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
        spinnerGender.setAdapter(gAdapter);
        adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, list);
        lv.setAdapter(adapter);
        refreshList();
        btnSave.setOnClickListener(v -> {
            String name = etName.getText().toString().trim();
            String gender = spinnerGender.getSelectedItem().toString();
            String addr = etAddress.getText().toString().trim();
            String nat = etNationality.getText().toString().trim();
            String mobile = etMobile.getText().toString().trim();

        });

        list.clear();

            String s = c.getString(0)+" | "+c.getString(1)+" | "+c.getString(2)+" | "+c.getString(3);
            list.add(s);
        }
        c.close();
        adapter.notifyDataSetChanged();
    }
```
