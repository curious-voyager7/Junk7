### Part-A: Background & Text Color Change

**MainActivity.java**

```java
package com.example.colorchange;

import android.os.Bundle;
import android.widget.*;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    Button btn;
    LinearLayout layout;
    TextView text;
    boolean toggle = true;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        layout = findViewById(R.id.layout);
        text = findViewById(R.id.text);
        btn = findViewById(R.id.btn);

        btn.setOnClickListener(v -> {
            if (toggle) {
                layout.setBackgroundColor(0xFF000000);
                text.setTextColor(0xFFFFFFFF);
            } else {
                layout.setBackgroundColor(0xFFFFFFFF);
                text.setTextColor(0xFF000000);
            }
            toggle = !toggle;
        });
    }
}
```


### Part-B: Product Entry (SQLite)

**MainActivity.java**

```java
package com.example.productentry;

import android.content.ContentValues;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.os.Bundle;
import android.widget.*;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    EditText pid, pname, price, qty, category;
    Button save, view;
    SQLiteDatabase db;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        pid = findViewById(R.id.pid);
        pname = findViewById(R.id.pname);
        price = findViewById(R.id.price);
        qty = findViewById(R.id.qty);
        category = findViewById(R.id.category);
        save = findViewById(R.id.save);
        view = findViewById(R.id.view);

        db = openOrCreateDatabase("ShopDB", MODE_PRIVATE, null);
        db.execSQL("CREATE TABLE IF NOT EXISTS products(ID INT, Name TEXT, Price INT, Quantity INT, Category TEXT)");

        save.setOnClickListener(v -> {
            ContentValues cv = new ContentValues();
            cv.put("ID", pid.getText().toString());
            cv.put("Name", pname.getText().toString());
            cv.put("Price", price.getText().toString());
            cv.put("Quantity", qty.getText().toString());
            cv.put("Category", category.getText().toString());
            db.insert("products", null, cv);
            Toast.makeText(this, "Saved", Toast.LENGTH_SHORT).show();
        });

        view.setOnClickListener(v -> {
            Cursor c = db.rawQuery("SELECT * FROM products", null);
            StringBuilder sb = new StringBuilder();
            while (c.moveToNext()) sb.append(c.getString(1)).append(" - ").append(c.getString(2)).append("\n");
            Toast.makeText(this, sb.toString(), Toast.LENGTH_LONG).show();
        });
    }
}
```

```java
package com.example.togglecolor;

import android.os.Bundle;
import android.widget.*;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    ToggleButton toggle;
    LinearLayout layout;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        layout = findViewById(R.id.layout);
```
